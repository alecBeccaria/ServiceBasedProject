//// Deploy network to Docker Container
// Create Network
docker network create netSEN300

// Deploy SQL server container
//	   name server         accept eula          environment variable pw    join network    set ports     name of the image
docker run --name sqlserver-b -e "ACCEPT_EULA=Y" -e "SA_PASSWORD=abc123!!!@" --net netSEN300 -p 1433:1433 -d mcr.microsoft.com/mssql/server:2019-latest

// Create Database on SqlServer
CREATE DATABASE catalogdb

// Create Table
USE catalogdb
CREATE TABLE "Items" (
	"Id" bigint NOT NULL PRIMARY KEY IDENTITY(1,1),
	"Title" VARCHAR(1000),
	"Description" VARCHAR(1000),
	"Price" DECIMAL(18,2),
	"Quantity" bigint
);
